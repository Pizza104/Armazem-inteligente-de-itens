-- Criar banco e tabelas
CREATE DATABASE IF NOT EXISTS empresa;
USE empresa;

-- Tabela cliente (sem telefone)
CREATE TABLE IF NOT EXISTS cliente (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE
);

-- Tabela fornecedor
CREATE TABLE IF NOT EXISTS fornecedor (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    cidade VARCHAR(100) NOT NULL
);

-- Procedures
DELIMITER $$

CREATE PROCEDURE sp_listar_clientes()
BEGIN
    SELECT id, nome, email FROM cliente ORDER BY id DESC;
END $$

CREATE PROCEDURE sp_inserir_cliente(IN p_nome VARCHAR(100), IN p_email VARCHAR(100))
BEGIN
    INSERT INTO cliente (nome, email) VALUES (p_nome, p_email);
END $$

CREATE PROCEDURE sp_listar_fornecedores()
BEGIN
    SELECT id, nome, cidade FROM fornecedor ORDER BY id DESC;
END $$

CREATE PROCEDURE sp_inserir_fornecedor(IN p_nome VARCHAR(100), IN p_cidade VARCHAR(100))
BEGIN
    INSERT INTO fornecedor (nome, cidade) VALUES (p_nome, p_cidade);
END $$

DELIMITER ;
